基本语法

添加注释

--这是单行注释

--[[这是多行注释
  这是多行注释
--]]

声明变量

--能在多个文件中调用
a = 10
--作用域仅为当前文件
local b = 10
--声明多个变量
c, d = 10, 20

输出语句

a = 10
print(a)
b, c, d = 10, 20, 30
print(b, c, d) --输出结果中每个变量中间都有一个'\t'

nil

--Lua里的nil类似于NULL或null
print(a) --输出未初始化的变量，输出结果为nil

number类型

a = 10       --10
b = 0x12     --18
c = 2e5      --200000.0
d = 2 ^ 4    --16
e = 1 << 3   --8
-- a, b, c, d, e 都是number类型
-- a++ a-- 不支持自增自减操作

字符串

a = "abcdas"
b = 'zzz'
c = "ab\nasd"     --'\n'会转换成一个换行
d = [[123
123\\\\n\t
]]                --可以打印出多行字符串，并完全保留字符串里的内容，即'\n'不会转换成换行 

e = a..b          --两个字符串的连接

f = tostring(123) --输出结果为字符串“123”

g = tonumber("13")--13
g = tonumber(abc) --输出结果为nil

print(#f)         --输出长度，3

s = string.char(0x30, 0x31, 0x32)  
print(s)                        --输出结果为012
n = string.byte(s, 2)           --输出结果为49，表示'1'

函数

function f(a, b)
    print(a, b, c)
    return a, b
end
--或
f = function(a, b)
-- ...
end

f(1, 2)          --打印结果为 1  2   nil
print(f(1, 2))   --函数返回结果为 1  2

i, j = f(1, 2)   --多重赋值
print(i, j)      --函数返回结果为 1  2

table（数组）

a = {1, "abc", {}, function() end}  --可以存任何类型

print(a[2])                         --下标从1开始，打印结果为"abc"
print(a[5])                         --未定义的，结果为nil

a[5] = 123                          
print(a[5])                         --此时可以打印出123

print(#a)                           --输出数组长度，5

table.insert(a, "abab")             --在a[6]插入
table.insert(a, 2, "abab")          --在a[2]插入，后面的元素后移一位
table.remove(a, 2)                  --删除a[2]，返回值为a[2]，后面的元素前移一位

用字符串当table下标

a = {
    a = 1,
    b = "123ab",
    c = function()

    end,
    d = {},
    [",;"] = 123
}                         --初始化
print(a[",;"])            --输出123
print(a.a)                --输出1
print(a["xyz"])           --输出nil

全局表_G

-- 本身是一个table，存储所有全局变量，即没有local修饰的变量
a = 1
print(_G["a"])             --1

print(_G["table"]["insert"])

布尔型

print(1 ~= 2)   --~=表示不等于

--在Lua里，只有false和nil代表假，0也代表真
                    a = true           a = 0   --真
                    b = false          b = nil --假

print(a and b)       false             nil
print(a or b)        true              0 
print(not a)         false             false

--and or 会返回a，b的值，而not不会受到影响
print(1 > 10 and "yes" or "no")                --输出结果为"no"

分支判断

if 1 > 10 then
    print("1 > 10")
elseif 1 > 5 then
    print("1 > 5")
else 
    print("1 < 10")
end

循环语句

for i = 1, 10 do              --打印 1 到 10，步长默认为 1
    print(i)
end

for i = 1, 10, 2 do           --打印 1 到 10，步长为 2
    print(i)
    i = 10                    --没用，在循环里不能修改 i 的值
    if i >= 5 then break end
end

n = 10
while n > 1 do

end
